SELECT s6.bar1 AS bar1, s6.bar2 AS bar2
FROM (
	SELECT *
	FROM (
		SELECT bar AS bar1
		FROM serves
	) AS s1, 
	(
		SELECT bar AS bar2
		FROM serves
	) AS s2
	WHERE NOT s1.bar1 = s2.bar2

	EXCEPT

	(SELECT DISTINCT s5.bar1, s5.bar2
	FROM (
	SELECT DISTINCT s1.bar1, s1.beer, s2.bar2
	FROM (
		SELECT bar AS bar1, beer
		FROM serves
	) AS s1, 
	(
		SELECT bar AS bar2
		FROM serves
	) AS s2
	WHERE NOT s1.bar1 = s2.bar2

	EXCEPT

	SELECT DISTINCT s3.bar, s3.beer, s4.bar
	FROM (
		SELECT *
		FROM serves
	) AS s3,
	(
		SELECT *
		FROM serves
	) AS s4
	WHERE s3.beer = s4.beer AND NOT s3.bar = s4.bar
	) AS s5) 
) AS s6

EXCEPT

SELECT s6.bar2 AS bar1, s6.bar1 AS bar2
FROM (
	SELECT *
	FROM (
		SELECT bar AS bar1
		FROM serves
	) AS s1, 
	(
		SELECT bar AS bar2
		FROM serves
	) AS s2
	WHERE NOT s1.bar1 = s2.bar2

	EXCEPT

	(SELECT DISTINCT s5.bar1, s5.bar2
	FROM (
	SELECT DISTINCT s1.bar1, s1.beer, s2.bar2
	FROM (
		SELECT bar AS bar1, beer
		FROM serves
	) AS s1, 
	(
		SELECT bar AS bar2
		FROM serves
	) AS s2
	WHERE NOT s1.bar1 = s2.bar2

	EXCEPT

	SELECT DISTINCT s3.bar, s3.beer, s4.bar
	FROM (
		SELECT *
		FROM serves
	) AS s3,
	(
		SELECT *
		FROM serves
	) AS s4
	WHERE s3.beer = s4.beer AND NOT s3.bar = s4.bar
	) AS s5) 
) AS s6
;
