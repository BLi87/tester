SELECT FIRST.BAR, SECOND.BAR FROM (SELECT BAR, array_agg(SERVES.BEER) AS BEERS FROM SERVES GROUP BY SERVES.BAR) AS FIRST, (SELECT BAR, array_agg(SERVES.BEER) AS BEERS FROM SERVES GROUP BY SERVES.BAR) AS SECOND WHERE FIRST.BEERS <@ SECOND.BEERS AND FIRST.BAR <> SECOND.BAR AND FIRST.BEERS<>SECOND.BEERS;
