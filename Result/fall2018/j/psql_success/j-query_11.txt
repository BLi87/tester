SELECT e.s2b, e.s1b FROM((SELECT c.s1b, c.s2b FROM ((SELECT s1.bar as s1b, s2.bar as s2b, s1.beer FROM Serves s1, Serves s2 WHERE s1.bar != s2.bar) EXCEPT (SELECT s1.bar as s1b, s2.bar as s2b, s1.beer FROM Serves s1, Serves s2 WHERE s1.bar != s2.bar AND s1.beer = s2.beer)) as c) EXCEPT (SELECT d.s1b, d.s2b FROM ((SELECT s2.bar as s1b, s1.bar as s2b, s1.beer FROM Serves s1, Serves s2 WHERE s1.bar != s2.bar) EXCEPT (SELECT s2.bar as s1b, s1.bar as s2b, s1.beer FROM Serves s1, Serves s2 WHERE s1.bar != s2.bar AND s1.beer = s2.beer)) as d))as e;
